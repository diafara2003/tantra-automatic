	public void cargarprocesos()
	{
		ListBox1.Items.Clear();
		processlist = Process.GetProcesses();
		Process[] array = processlist;
		foreach (Process process in array)
		{
			if (!string.IsNullOrEmpty(process.MainWindowTitle))
			{
				if (Operators.CompareString(process.MainWindowTitle.ToString(), " ", TextCompare: false) == 0)
				{
					SetWindowText(process.MainWindowHandle, "HTLauncher");
				}
				ListBox1.Items.Add(process.MainWindowTitle.ToString());
			}
		}
	}

	private void CambiarNombreProceso(object sender, EventArgs e)
	{
		if (!string.IsNullOrEmpty(seleccionado))
		{
			if (!string.IsNullOrEmpty(TextBox1.Text))
			{
				if (TextBox1.Text.Length <= 20)
				{
					Process[] array = processlist;
					foreach (Process process in array)
					{
						if (!string.IsNullOrEmpty(process.MainWindowTitle) & (Operators.CompareString(process.MainWindowTitle.ToString(), seleccionado, TextCompare: false) == 0) & (process.Id == seleccionadoID))
						{
							SetWindowText(process.MainWindowHandle, TextBox1.Text);
							cargarprocesos();
							Text = TextBox1.Text;
						}
					}
				}
				else
				{
					Interaction.MsgBox("Enter a short name");
				}
			}
			else
			{
				Interaction.MsgBox("Enter a valid name");
			}
		}
		else
		{
			Interaction.MsgBox("Select a process from the list");
		}
	}

	private void seleccionarProceso(object sender, EventArgs e)
	{
		if (ListBox1.SelectedIndex < 0)
		{
			return;
		}
		seleccionado = ListBox1.GetItemText(RuntimeHelpers.GetObjectValue(ListBox1.SelectedItem));
		TextBox1.Text = seleccionado;
		TextBox1.Select();
		Process[] array = processlist;
		foreach (Process process in array)
		{
			if (!string.IsNullOrEmpty(process.MainWindowTitle) & (Operators.CompareString(process.MainWindowTitle.ToString(), seleccionado, TextCompare: false) == 0))
			{
				proceso = process;
				seleccionadoID = proceso.Id;
				try
				{
					handle = FindWindow(null, seleccionado);
				}
				catch (Exception ex)
				{
					ProjectData.SetProjectError(ex);
					Exception ex2 = ex;
					ProjectData.ClearProjectError();
				}
			}
		}
	}

	public Color GetPixelColor(int x, int y)
	{
		IntPtr windowDC = GetWindowDC(IntPtr.Zero);
		uint pixel = GetPixel(windowDC, x, y);
		ReleaseDC(IntPtr.Zero, windowDC);
		checked
		{
			return Color.FromArgb((int)Conversion.Int(unchecked((long)pixel) & 0xFFL), (int)(Conversion.Int(unchecked((long)pixel) & 0xFF00L) >> 8), (int)(Conversion.Int(unchecked((long)pixel) & 0xFF0000L) >> 16));
		}
	}

	public bool EsCasiNegro(Color color, int tolerancia = 30)
	{
		if (color.R < tolerancia && color.G < tolerancia)
		{
			return color.B < tolerancia;
		}
		return false;
	}

	private void guardararchivo(object sender, EventArgs e)
	{
		try
		{
			if (Operators.CompareString(TextBox1.Text, string.Empty, TextCompare: false) == 0)
			{
				Interaction.MsgBox("You must specify a name for the file to be saved.", MsgBoxStyle.Information, "Save");
				return;
			}
			if (File.Exists(ruta))
			{
				File.Create(ruta + TextBox1.Text + ".txt").Close();
			}
			else
			{
				Directory.CreateDirectory(ruta);
				File.Create(ruta + TextBox1.Text + ".txt").Close();
			}
			StreamWriter streamWriter = new StreamWriter(ruta + TextBox1.Text + ".txt");
			try
			{
				string value = obtenerdata();
				streamWriter.WriteLine(value);
				streamWriter.Close();
				Interaction.MsgBox("File " + TextBox1.Text + " created successfully", MsgBoxStyle.Information, "Save Presets");
			}
			catch (Exception ex)
			{
				ProjectData.SetProjectError(ex);
				Exception ex2 = ex;
				Interaction.MsgBox("There is a problem writing to the file: " + ex2.Message, MsgBoxStyle.Critical, "Save Presets");
				ProjectData.ClearProjectError();
			}
		}
		catch (Exception ex3)
		{
			ProjectData.SetProjectError(ex3);
			Exception ex4 = ex3;
			Interaction.MsgBox("There is a problem writing to the file: " + ex4.Message, MsgBoxStyle.Critical, "Save Presets");
			ProjectData.ClearProjectError();
		}
	}

		{
			Directory.CreateDirectory(ruta);
		}

		if (openFileDialog.ShowDialog() != DialogResult.OK)
		{
			return;
